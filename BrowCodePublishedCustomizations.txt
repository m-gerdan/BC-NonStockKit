  {
    
    public PXSelect<MABZGLAccountNumberUpdate> AccountRecords;
    public PXAction<BZ.MABZGLAccountNumberUpdate> ProcessUpdates;
  
    [PXButton(CommitChanges = true)]
    [PXUIField(DisplayName = "Process Updates")]
    protected void processUpdates()
    {
    
      foreach (MABZGLAccountNumberUpdate rec in AccountRecords.Select())
      {
    
        if (string.Compare(rec.OldAccountCD, rec.NewAccountCD, true) != 0)
        {
    
          Account a = new PXSelectReadonly<Account, Where<Account.accountCD, Equal<Required<Account.accountCD>>>>(this).Select(rec.OldAccountCD);
          if (a != null)
          {
    
            PXDatabase.Update<Account>(
              new PXDataFieldAssign<Account.accountCD>(rec.NewAccountCD),
              new PXDataFieldRestrict<Account.accountID>(a.AccountID));

          }    
    
        }
    
      }

    }

CREATE OR REPLACE VIEW MABrowCodeKitCountMax As
SELECT
  InventoryItem.CompanyID,
  InventoryItem.InventoryID,
  InventoryItem.InventoryCD,
  InventoryItem.ItemStatus,
  InventoryItem.ItemClassID,
  InventoryItem.Descr,
  InventoryItem.BaseUnit,
  FLOOR(Min(
    CASE WHEN INKitSpecStkDet.DfltCompQty > 0 THEN INSiteStatus.QtyAvail / INKitSpecStkDet.DfltCompQty ELSE 0 END
  )) AS MaxKitCount
FROM
  INKitSpecStkDet as INKitSpecStkDet,
  INKitSpecHdr as INKitSpecHeader,
  InventoryItem as InventoryItem,
  INSiteStatus as INSiteStatus
WHERE
  1 = 1
  AND InventoryItem.InventoryID = INSiteStatus.InventoryID
  AND INKitSpecStkDet.CompInventoryID = InventoryItem.InventoryID
  AND INKitSpecStkDet.KitInventoryID = INKitSpecHeader.KitInventoryID
  AND INKitSpecStkDet.CompanyID = INKitSpecHeader.CompanyID
  AND INKitSpecStkDet.CompanyID = InventoryItem.CompanyID
  AND INKitSpecStkDet.CompanyID = INSiteStatus.CompanyID
GROUP BY
  INKitSpecStkDet.KitInventoryID;

#GET NON KIT INVENTORY

  InventoryItem.Descr,
  InventoryItem.BaseUnit,
  INSiteStatus.QtyAvail AS MaxKitCount
FROM
  InventoryItem as InventoryItem,
  INSiteStatus as INSiteStatus
WHERE
  1 = 1
  AND InventoryItem.InventoryID = INSiteStatus.InventoryID
  AND InventoryItem.CompanyID = INSiteStatus.CompanyID
GROUP BY
  InventoryItem.InventoryID;]]></CDATA>
